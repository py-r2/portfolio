def render_markdown(path):
    with codecs.open(path, 'r', 'utf-8') as md_file:
        md_text = md_file.read()
    md = markdown.Markdown(
        extensions=['meta'],
        output_format='html5')
    html = md.convert(md_text)
    meta = {key: value[0] for key, value in md.Meta.items()}
    return meta, html
